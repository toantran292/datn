\subsection{Các yêu cầu phi chức năng}

\subsubsection{Yêu cầu thực thi}
\begin{itemize}
    \item Thiết bị sử dụng cần kết nối Internet ổn định.
    \item Thời gian phản hồi yêu cầu không vượt quá 3 giây.
    \item Board view và Backlog view load trong vòng 2 giây với tối đa 500 issues.
    \item Drag-and-drop issue cập nhật real-time, độ trễ dưới 500ms.
    \item Tìm kiếm và filter issues trả kết quả trong 1 giây.
    \item Tính toán sprint metrics (velocity, completion rate) trong 2 giây.
    \item Ứng dụng hoạt động ổn định, không lỗi nghiêm trọng.
    \item Hỗ trợ tối thiểu 100 người dùng đồng thời cập nhật issues trong cùng một project.
    \item AI Sprint Summary generation hoàn thành trong 30 giây.
    \item AI Issue Description và Task Breakdown hoàn thành trong 15 giây.
\end{itemize}

\subsubsection{Yêu cầu an toàn}
\begin{itemize}
    \item Đảm bảo sao lưu và khôi phục dữ liệu an toàn khi nâng cấp hoặc bảo trì.
    \item Khả năng mở rộng để xử lý lượng lớn dữ liệu: projects, sprints, issues, comments, activities.
    \item Hệ thống ổn định, không chứa mã độc hoặc phần mềm độc hại.
    \item Backup tự động hàng ngày, lưu trữ tối thiểu 30 ngày.
    \item Đảm bảo tính toàn vẹn dữ liệu (data integrity) khi cascade delete project hoặc sprint.
    \item Transaction safety cho các thao tác quan trọng: complete sprint, move issues, delete project.
    \item Xử lý concurrent updates (optimistic locking) để tránh mất dữ liệu khi nhiều user edit cùng issue.
    \item Mã hóa dữ liệu khi truyền tải (TLS 1.3) và lưu trữ (AES-256).
\end{itemize}

\subsubsection{Yêu cầu về giao diện}
\begin{itemize}
    \item Giao diện hiện đại, màu sắc hài hòa, phong cách chuyên nghiệp phù hợp doanh nghiệp.
    \item Board view hiển thị status columns rõ ràng với issues dạng cards.
    \item Backlog view hiển thị sprints (ACTIVE, FUTURE, CLOSED) và backlog section có thể expand/collapse.
    \item Issue detail panel slide-in từ bên phải với tabs: Details, Activity, Comments.
    \item Drag-and-drop UX mượt mà với visual feedback (drag preview, drop zone highlight).
    \item Custom status colors hiển thị consistent trên badges, columns, và issue cards.
    \item Sprint progress bar và metrics (velocity, completion rate) hiển thị trực quan.
    \item Filter panel với multi-select cho assignee, priority, type, status.
    \item AI-generated content (sprint summary, issue description) hiển thị với formatting rõ ràng.
    \item Đảm bảo nhất quán kiểu chữ, biểu tượng và màu sắc theo design system.
    \item Hỗ trợ Dark Mode và Light Mode.
    \item Responsive design cho tablet và mobile (read-only mode).
\end{itemize}

\subsubsection{Yêu cầu về tính bảo mật}
\begin{itemize}
    \item Phân quyền theo vai trò trong project: Project Lead (full access), Team Member (create/update issues), Viewer (read-only).
    \item Project-level access control: User chỉ xem/edit được projects thuộc organization của mình.
    \item Issue visibility control: Private issues chỉ visible cho assignees và Project Lead.
    \item Activity logging: Ghi lại tất cả thay đổi quan trọng (create/update/delete project, sprint, issue).
    \item Audit trail: Track who created/modified/deleted với timestamp để investigation.
    \item Mã hóa mật khẩu an toàn (bcrypt với salt) trước khi lưu CSDL.
    \item Tuân thủ quy định bảo vệ dữ liệu cá nhân như GDPR.
    \item Hỗ trợ xác thực hai yếu tố (2FA) tùy chọn.
    \item Tích hợp Google OAuth tuân thủ OAuth 2.0 và OpenID Connect.
    \item Session token có thời hạn, làm mới định kỳ, hỗ trợ đăng xuất tất cả thiết bị.
    \item API bảo vệ bằng JWT với thời hạn ngắn (15 phút) và refresh token (7 ngày).
    \item Rate limiting cho API endpoints để tránh abuse: 100 requests/phút/user.
    \item Input validation và sanitization để tránh XSS, SQL Injection.
\end{itemize}
