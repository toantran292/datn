Luồng xử lý bình thường &
\begin{minipage}{\linewidth}
    \vskip 4pt
    \begin{enumerate}
        \item Member truy cập vào trang danh sách kênh của workspace.
        \item Hệ thống hiển thị danh sách kênh: kênh đang tham gia và kênh công khai có thể tham gia.
        \item Member chọn một kênh công khai chưa tham gia.
        \item Member nhấn "Tham gia kênh".
        \item Hệ thống thêm Member vào kênh.
        \item Member có thể xem và gửi tin nhắn trong kênh.
    \end{enumerate}
    \vskip 1pt
\end{minipage}
