Luồng xử lý bình thường &
\begin{minipage}[t]{\linewidth}
\vspace{2pt}
\begin{enumerate}[nosep, leftmargin=*]
    \item Super Admin truy cập trang quản trị work\-space.
    \item Super Admin chọn work\-space cần thay đổi trạng thái.
    \item Super Admin chọn "Khóa" hoặc "Mở khóa".
    \item Nếu khóa: Nhập lý do khóa work\-space.
    \item Hệ thống cập nhật trạng thái work\-space.
    \item Hệ thống gửi thông báo đến tất cả thành viên.
    \item Hệ thống ghi audit log.
\end{enumerate}
\vspace{2pt}
\end{minipage} \\
Các luồng sự kiện con &
\begin{minipage}[t]{\linewidth}
\vspace{2pt}
\begin{itemize}[nosep, leftmargin=*]
    \item Khóa Work\-space: Ngăn không cho thành viên truy cập và thao tác.
    \item Mở khóa Work\-space: Khôi phục quyền truy cập bình thường.
\end{itemize}
