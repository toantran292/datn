\subsection{Các yêu cầu phi chức năng}

\subsubsection{Yêu cầu thực thi}
\begin{itemize}
    \item Thiết bị sử dụng cần kết nối mạng ổn định.
    \item Thời gian phản hồi yêu cầu không vượt quá 3 giây.
    \item Kết quả từ chức năng tìm kiếm, xem không gian làm việc, xem báo cáo trí tuệ nhân tạo đảm bảo chính xác.
    \item Ứng dụng hoạt động ổn định, không lỗi nghiêm trọng.
    \item Hỗ trợ tối thiểu 1000 người dùng đồng thời.
    \item Thời gian xử lý báo cáo trí tuệ nhân tạo không quá 30 giây.
\end{itemize}

\subsubsection{Yêu cầu an toàn}
\begin{itemize}
    \item Đảm bảo sao lưu và khôi phục dữ liệu an toàn khi nâng cấp hoặc bảo trì.
    \item Khả năng mở rộng để xử lý lượng lớn dữ liệu không gian làm việc, tệp tin, thành viên.
    \item Hệ thống ổn định, không chứa mã độc hoặc phần mềm độc hại.
    \item Sao lưu tự động hàng ngày, lưu trữ tối thiểu 30 ngày.
    \item Mã hóa dữ liệu khi truyền tải và lưu trữ.
\end{itemize}

\subsubsection{Yêu cầu về giao diện}
\begin{itemize}
    \item Giao diện hiện đại, màu sắc hài hòa, phong cách chuyên nghiệp phù hợp doanh nghiệp.
    \item Bố cục điều hướng, không gian làm việc, tệp tin, bảng điều khiển, báo cáo trí tuệ nhân tạo rõ ràng, dễ thao tác.
    \item Hiển thị báo cáo trí tuệ nhân tạo trực quan với các phân tích, đề xuất hành động và xuất báo cáo mượt mà.
    \item Đảm bảo nhất quán kiểu chữ, biểu tượng và màu sắc.
    \item Hỗ trợ chế độ tối và chế độ sáng.
\end{itemize}

\subsubsection{Yêu cầu về tính bảo mật}
\begin{itemize}
    \item Phân quyền theo vai trò (Quản trị viên hệ thống, Chủ không gian làm việc, Thành viên).
    \item Mã hóa mật khẩu an toàn trước khi lưu vào cơ sở dữ liệu.
    \item Tuân thủ quy định bảo vệ dữ liệu cá nhân.
    \item Hỗ trợ xác thực hai yếu tố tùy chọn.
    \item Tích hợp đăng nhập qua tài khoản bên thứ ba tuân thủ các chuẩn bảo mật phổ biến.
    \item Mã thông báo phiên làm việc có thời hạn, làm mới định kỳ, hỗ trợ đăng xuất tất cả thiết bị.
    \item Giao diện lập trình được bảo vệ với thời hạn ngắn và cơ chế làm mới.
\end{itemize}
